char szBinary[16];

char * dec2bin(int dec)
{
        int working = dec;
        int temp;
        memset(szBinary,0,sizeof(szBinary));

        for(int i=0;i<8;i++) {
                if(working & 1) {
                        szBinary[7-i] = '1';                    
                } else {
                        szBinary[7-i] = '0';
                }       
                temp = working;
                working = temp >> 1;                    
        }
        return szBinary;
}
