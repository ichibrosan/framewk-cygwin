#include "asdefs.hpp"
#define DWGALLOC
#define INIBIND
#define MEMORY
#include "includes.hpp"

int main(int argc,char **argv)
{
        Version(argv[0]);

        if(3 != argc) {
                printf("usage sb <address> <value>\n");
                exit(EXIT_FAILURE);
        }

        int offset,value;
        sscanf(argv[1],"%04X",&offset);
        sscanf(argv[2],"%02X",&value);

        CMemory mem(MEMSIZE);
        mem.pokeb(offset,value);
}
