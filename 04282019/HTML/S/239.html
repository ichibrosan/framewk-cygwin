                        case 'C':
                        case 'c': g_iSegment = SEG_CODE; break;

                        case 'd': g_iSegment = SEG_DATA; break;
                        case 'S':
                        case 's': g_iSegment = SEG_STACK; break;
                        default:  g_iSegment = SEG_VAGUE; break;
                }
        } else {
                g_iErrorNo |= ERROR_NOPARM;
