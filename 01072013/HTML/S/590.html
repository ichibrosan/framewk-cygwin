#include "asdefs.hpp"
#define DWGALLOC
#define INIBIND
#define MEMORY
#include "includes.hpp"

int main(int argc,char **argv)
{
        Version(argv[0]);

        if(2 != argc) {
                printf("usage - d <address>\n");
                exit(EXIT_FAILURE);
        }

        int offset;
        sscanf(argv[1],"%04X",&offset);

        printf("%04X: ",offset);        
        CMemory mem(MEMSIZE);
        for(int i=0;i<16;i++) {
                printf("%02X ",mem.peekb(offset+i));
        }
        printf("\n");
}
